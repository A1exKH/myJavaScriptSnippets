/*

Task1

14. Заданы два массива А(5) и В(5). В каждом из массивов найти наименьшее значение и умножить на него все элементы массивов. На печать вывести исходные и преобразованные массивы.

*/

int main(){

    //variables

    int a = 0;

    int b = 0;



	cout<<"Vvedite chislo elementov massiva A: ";

	cin>>a;





    cout<<"Vvedite chislo elementov massiva B: ";

    cin>>b;



    int *A = new int[a];

    int *B = new int[b];



    //input massives

    for(int i=0; i<a;i++){

        cout<<"Vvedite A["<<i<<"] = ";

        cin>>A[i];



    }



    for(int i=0; i<b;i++){

        cout<<"Vvedite B["<<i<<"] = ";

        cin>>B[i];



    }



    int minValueA = A[0];

    int minValueB = B[0];



    //output old massives

    cout<<"Old values of massives:"<<endl;



    for(int i=0;i<a;i++){

        cout<<"A["<<i<<"] = "<<A[i]<<endl;

    }



    for(int i=0;i<b;i++){

        cout<<"B["<<i<<"] = "<<B[i]<<endl;

    }



    //find min value of each massive

    for(int i=0;i<a;i++){

        if(A[i]<minValueA){

            minValueA = A[i];

        }

    }



    for(int i=0;i<b;i++){

        if(B[i]<minValueB){

            minValueB = B[i];

        }

    }



    //multiply each massive value on min value

    for(int i=0;i<a;i++){

        A[i] *= minValueA ;

    }



    for(int i=0;i<b;i++){

        B[i] *= minValueB ;

    }

    //output massives

    cout<<"New values of massives:"<<endl;



    for(int i=0;i<a;i++){

        cout<<"A["<<i<<"] = "<<A[i]<<endl;

    }



    for(int i=0;i<b;i++){

        cout<<"B["<<i<<"] = "<<B[i]<<endl;

    }



    getch();

    return 0;

}
